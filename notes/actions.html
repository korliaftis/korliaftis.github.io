on:
  workflow_dispatch:
    inputs:
      uuid:
        description: "uuid"
        required: false
        type: string
      outcome:
        description: "outcome"
        required: true
        type: choice
        default: "success"
        options:
        - success
        - failure

jobs:

  main:
    runs-on: ubuntu-latest

        if [[ "${{ inputs.outcome }}" == "failure" ]]; then

#!/bin/bash

set -e
set -u
set -o pipefail

VERY_SECRET_TOKEN="" # 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
REPOSITORY_NAME=""   # stuff
REPOSITORY_OWNER=""  # korliaftis
WORKFLOW_BRANCH=""   # master
WORKFLOW_FILENAME="" # target.yaml
WORKFLOW_OUTCOME=""  # success | failure

WORKFLOW_UUID=$(uuidgen)

curl -L -s --fail-with-body \
  -X POST \
  -H "Accept: application/vnd.github+json" \
  -H "Authorization: Bearer ${VERY_SECRET_TOKEN}" \
  -H "X-GitHub-Api-Version: 2022-11-28" \
  "https://api.github.com/repos/${REPOSITORY_OWNER}/${REPOSITORY_NAME}/actions/workflows/${WORKFLOW_FILENAME}/dispatches" \
  -d "{
      \"ref\": \"${WORKFLOW_BRANCH}\",
      \"inputs\": {
        \"uuid\": \"${WORKFLOW_UUID}\",
        \"outcome\": \"${WORKFLOW_OUTCOME}\"
      }
  }"

sleep 10

WORKFLOW_RUN_ID=$(curl -L -s --fail-with-body \
  -H "Accept: application/vnd.github+json" \
  -H "Authorization: Bearer ${VERY_SECRET_TOKEN}" \
  -H "X-GitHub-Api-Version: 2022-11-28" \
  "https://api.github.com/repos/${REPOSITORY_OWNER}/${REPOSITORY_NAME}/actions/workflows/${WORKFLOW_FILENAME}/runs" \
  | jq -r --arg rn "${WORKFLOW_UUID}" '.workflow_runs[] | select(.display_title==$rn) | .id' | head -n 1)

echo "id: ${WORKFLOW_RUN_ID}"

echo "url: https://github.com/${REPOSITORY_OWNER}/${REPOSITORY_NAME}/actions/runs/${WORKFLOW_RUN_ID}"

for i in {1..300}; do

  WORKFLOW_RUN_STATUS=$(curl -L -s --fail-with-body \
    -H "Accept: application/vnd.github+json" \
    -H "Authorization: Bearer ${VERY_SECRET_TOKEN}" \
    -H "X-GitHub-Api-Version: 2022-11-28" \
    "https://api.github.com/repos/${REPOSITORY_OWNER}/${REPOSITORY_NAME}/actions/runs/${WORKFLOW_RUN_ID}" \
    | jq -r '.status')

  echo "status: ${WORKFLOW_RUN_STATUS}"

  if [[ "${WORKFLOW_RUN_STATUS}" == "completed" ]]; then

    WORKFLOW_RUN_CONCLUSION=$(curl -L -s --fail-with-body \
      -H "Accept: application/vnd.github+json" \
      -H "Authorization: Bearer ${VERY_SECRET_TOKEN}" \
      -H "X-GitHub-Api-Version: 2022-11-28" \
      "https://api.github.com/repos/${REPOSITORY_OWNER}/${REPOSITORY_NAME}/actions/runs/${WORKFLOW_RUN_ID}" \
      | jq -r '.conclusion')

    echo "conclusion: ${WORKFLOW_RUN_CONCLUSION}"

    if [[ "${WORKFLOW_RUN_CONCLUSION}" == "success" ]]; then
      exit 0
    else
      exit 1
    fi

  fi

  sleep 5

done
